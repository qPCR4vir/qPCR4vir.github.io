{

nana::menubar mn(fm);

mn.at(0).append("open modal window", [&fm](nana::menu::item_proxy& ip)
{
nana::form fm2(fm);
fm2.show();
fm2.modality();
});
pl.collocate();
fm.show();
nana::exec();
return 0;
}
